### C++ Code 
```c++
#include <bits/stdc++.h> 
using namespace std;
int findAns(vector<int> arr, int n, int sum, int m)
{
    int start = 0;
    int end = sum;
    
    int ans = -1;
    int mid = start+(end-start)/2;
    while (start <= end)
    {

        int totalsum = 0;
        bool isPossible = true;
        int count = 0;
        for (int i = 0; i < n; i++)
        {
            if (count < m)
            {
                if ((totalsum+arr[i]) <= mid)
                {
                    totalsum += arr[i];
                }
                else
                {
                    count++;
                    --i;
                    totalsum = 0;
                }
            }
            else
            {
                start = mid + 1;
                isPossible =false;
                break;
            }
        }
        
        if (isPossible)
        {
            ans = mid;
            end = mid -1;
        }
        mid = start + (end - start) / 2;

    }
    return ans;
}
int allocateBooks(vector<int> arr, int n, int m) {
    
    int ans =0 ;
    int sum = 0;
    for (int i = 0; i < n; i++)
    {
        sum += arr[i];
    }
    ans = findAns(arr, n, sum, m);
    return ans;
}
```
