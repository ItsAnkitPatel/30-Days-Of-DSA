### C++ Code

```c++
class Solution {
    int doBinarySearch(int target)
{
    int ans = 0;
    int start = 0;
    int end = target;
    long long int mid = start + (end - start) / 2;
    while (start <= end)
    {
        if (mid * mid == target)
        {
            return mid;
        }
        else if (mid * mid < target)
        {
            start = mid + 1;
            ans = mid;
        }
        else
        {
            end = mid - 1;
        }
        mid = start + (end - start) / 2;
    }
    return ans;
}

public:
    int mySqrt(int x) {
        return doBinarySearch(x);
    }
};
```
