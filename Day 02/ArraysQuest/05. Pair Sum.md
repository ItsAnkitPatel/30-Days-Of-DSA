public static List<int[]> pairSum(int[] arr, int s) {
        Arrays.sort(arr);
        List<int[]> li = new ArrayList<>();
        for(int i =0; i <arr.length-1; i++){
            int element = arr[i];
            for(int j =i+1; j<arr.length; j++){
                if((element+arr[j])==s){
                    li.add(new int[]{element,arr[j]});
                }
            }
        }
        return li;
    }
